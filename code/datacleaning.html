# LOADING LIBRARIES
import pandas as pd
import datetime as dt
import wbdata
# 1) CLEAN ACLED EVENT-LEVEL DATA -> COUNTRY-MONTH PANEL

df = pd.read_csv("../data/ACLEDdatauncleaned.csv")

latin_america_countries = [
    "Argentina","Bolivia","Brazil","Chile","Colombia","Costa Rica","Cuba",
    "Dominican Republic","Ecuador","El Salvador","Guatemala","Haiti","Honduras",
    "Mexico","Nicaragua","Panama","Paraguay","Peru","Uruguay","Venezuela"
]

df["event_date"] = pd.to_datetime(df["event_date"], errors="coerce")
df = df.dropna(subset=["event_date"])

df = df[(df["event_date"].dt.year >= 2015) & (df["event_date"].dt.year <= 2025)].copy()

df["year_month"] = df["event_date"].dt.to_period("M").astype(str)

df["country"] = (df["country"].astype(str)
                 .str.replace("\u00a0", " ", regex=False)
                 .str.replace(r"\s+", " ", regex=True)
                 .str.strip())

df = df[df["country"].isin(latin_america_countries)].copy()

df["is_protest"] = (df["event_type"] == "Protests").astype(int)
df["is_escalation_event"] = df["event_type"].isin(["Riots", "Violence against civilians"]).astype(int)

panel = (
    df.groupby(["country", "year_month"], as_index=False)
      .agg(
          protest_count=("is_protest", "sum"),
          protest_any=("is_protest", "max"),
          violence_count=("is_escalation_event", "sum"),
          violence_any=("is_escalation_event", "max"),
          fatalities=("fatalities", "sum"),
          protest_days=("event_date", lambda x: x[df.loc[x.index, "is_protest"].eq(1)].nunique())
      )
)


panel["year_month"] = pd.PeriodIndex(panel["year_month"], freq="M")
panel = panel.sort_values(["country", "year_month"]).reset_index(drop=True)

panel["violence_any_t1"] = panel.groupby("country")["violence_any"].shift(-1)
panel["violence_count_t1"] = panel.groupby("country")["violence_count"].shift(-1)

panel["escalation"] = ((panel["protest_any"] == 1) & (panel["violence_any_t1"] == 1)).astype(int)

# Restrict to protest months only (your design)
panel = panel[panel["protest_any"] == 1].copy()

panel["protest_count_t_1"] = panel.groupby("country")["protest_count"].shift(1)
panel["protest_growth"] = panel["protest_count"] - panel["protest_count_t_1"]
panel["recent_violence"] = (
    (panel["violence_any"] == 1) | (panel.groupby("country")["violence_any"].shift(1) == 1)
).astype(int)

panel["year"] = panel["year_month"].dt.year.astype(int)

# Clean country strings (ACLED)
panel["country"] = (panel["country"].astype(str)
                    .str.replace("\u00a0", " ", regex=False)
                    .str.replace(r"\s+", " ", regex=True)
                    .str.strip())
# Sort properly
panel = panel.sort_values(["country", "year_month"])

# Create country-level means
country_means = panel.groupby("country").transform("mean")

# Demean dynamic predictors
panel["protest_count_dm"] = panel["protest_count"] - country_means["protest_count"]
panel["protest_growth_dm"] = panel["protest_growth"] - country_means["protest_growth"]
panel["recent_violence_dm"] = panel["recent_violence"] - country_means["recent_violence"]
panel["protest_count_t_2"] = panel.groupby("country")["protest_count"].shift(2)

panel["protest_acceleration"] = (
    panel["protest_count"] 
    - 2 * panel["protest_count_t_1"] 
    + panel["protest_count_t_2"]
)
variation = panel.groupby("country")["escalation"].nunique()

countries_with_variation = variation[variation > 1].index

panel_model = panel[panel["country"].isin(countries_with_variation)]
# 2) CLEAN DEMOCRACY INDEX DATA (LATIN AMERICA YEAR >= 2018)
df3 = pd.read_csv("../data/democracyindex.csv")
df3.columns = df3.columns.str.strip()

dem = df3[(df3["Year"] >= 2015) & (df3["Year"] <= 2025)].copy()

dem = dem.rename(columns={
    "Entity": "country",
    "Year": "year",
    "Democracy Index": "democracy_index"
})

dem["year"] = dem["year"].astype(int)

dem["country"] = (dem["country"].astype(str)
                  .str.replace("\u00a0", " ", regex=False)
                  .str.replace(r"\s+", " ", regex=True)
                  .str.strip())

# KEEP ONLY LATAM COUNTRIES
dem = dem[dem["country"].isin(latin_america_countries)].copy()
# 3) MERGE DEMOCRACY ONTO FILTERED PANEL
panel2 = panel.merge(
    dem[["country", "year", "democracy_index"]],
    on=["country", "year"],
    how="left"
)

print("Panel rows (protest months only):", len(panel2))
print("Escalations:", int(panel2["escalation"].sum()))
print("Escalation rate:", round(panel2["escalation"].mean() * 100, 2), "%")
print("Missing democracy_index after merge:", int(panel2["democracy_index"].isna().sum()))

# Convert year_month back to string & save
panel2["year_month"] = panel2["year_month"].astype(str)
panel2.to_csv("../data/ACLEDcleaned.csv", index=False)
